Imports System.Configuration
Imports System.Collections
Imports System.Web.Security
Imports System.Web.UI
Imports System.Web.UI.WebControls
Imports System.Web.UI.WebControls.WebParts
Imports System.Web.UI.HtmlControls

Imports HiQPdf

Namespace HiQPdf_Demo
    Partial Public Class FillAndSaveForms
        Inherits System.Web.UI.Page
        Protected Sub buttonFillAndSavePdf_Click(ByVal sender As Object, ByVal e As EventArgs)
            Dim pdfFileWithForm As String = Server.MapPath("~") & "\DemoFiles\Pdf\PdfWithForm.pdf"

            ' load the PDF document with form from file
            Dim document As PdfDocument = PdfDocument.FromFile(pdfFileWithForm)

            ' set a demo serial number
            document.SerialNumber = "YCgJMTAE-BiwJAhIB-EhlWTlBA-UEBRQFBA-U1FOUVJO-WVlZWQ=="

            ' get the Check Box field by name from form fields collection and set its value
            Dim checkBoxField As PdfFormField = document.Form.Fields("cb")
            If checkBoxField IsNot Nothing Then
                checkBoxField.Value = checkBoxChecked.Checked
            End If

            ' get the Text Box field by name from form fields collection and set its value
            Dim textBoxField As PdfFormField = document.Form.Fields("tb")
            If textBoxField IsNot Nothing Then
                textBoxField.Value = textBoxText.Text
            End If

            ' get the List Box field by name from form fields collection and set its value
            Dim listBoxField As PdfFormField = document.Form.Fields("lb")
            If listBoxField IsNot Nothing Then
                listBoxField.Value = dropDownListListBoxValue.SelectedValue
            End If

            ' get the Combo Box field by name from form fields collection and set its value
            Dim comboBoxField As PdfFormField = document.Form.Fields("combo")
            If comboBoxField IsNot Nothing Then
                comboBoxField.Value = dropDownListComboBoxValue.SelectedValue
            End If

            ' get the Radio Buttons Group field by name from form fields collection and set its value
            Dim radioGroupField As PdfFormField = document.Form.Fields("rg")
            If radioGroupField IsNot Nothing Then
                If (radioButton1.Checked) Then
                    radioGroupField.Value = "rb1"
                Else
                    radioGroupField.Value = "rb2"
                End If
            End If

            Try
                ' write the PDF document to a memory buffer
                Dim pdfBuffer() As Byte = document.WriteToMemory()

                ' inform the browser about the binary data format
                HttpContext.Current.Response.AddHeader("Content-Type", "application/pdf")

                ' let the browser know how to open the PDF document and the file name
                HttpContext.Current.Response.AddHeader("Content-Disposition", String.Format("attachment; filename=FillForms.pdf; size={0}", pdfBuffer.Length.ToString()))

                ' write the PDF buffer to HTTP response
                HttpContext.Current.Response.BinaryWrite(pdfBuffer)

                ' call End() method of HTTP response to stop ASP.NET page processing
                HttpContext.Current.Response.End()
            Finally
                document.Close()
            End Try
        End Sub

        Protected Sub Page_Load(ByVal sender As Object, ByVal e As EventArgs)
            If Not IsPostBack Then
                Dim pageUri As String = HttpContext.Current.Request.Url.AbsoluteUri
                hyperLinkOpen.NavigateUrl = pageUri.Substring(0, pageUri.LastIndexOf("/"c)) & "/DemoFiles/Pdf/PdfWithForm.pdf"
                textBoxPdf.Text = hyperLinkOpen.NavigateUrl

                dropDownListListBoxValue.SelectedValue = "List Value 2"
                dropDownListComboBoxValue.SelectedValue = "Combo Value 2"

                Master.SelectNode("fillAndSaveForms")
            End If
        End Sub
    End Class
End Namespace
