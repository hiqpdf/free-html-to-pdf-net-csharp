Imports HiQPdf

Public Class SearchTextInPdf
    Inherits System.Web.UI.Page

    Protected Sub buttonSearchText_Click(sender As Object, e As EventArgs) Handles buttonSearchText.Click
        ' get the PDF file
        Dim pdfFile As String = Server.MapPath("~") + "\DemoFiles\Pdf\InputPdf.pdf"

        ' get the text to search
        Dim textToSearch As String = textBoxTextToSearch.Text

        ' create the PDF text extractor
        Dim pdfTextExtractor As PdfTextExtract = New PdfTextExtract()

        ' set a demo serial number
        pdfTextExtractor.SerialNumber = "YCgJMTAE-BiwJAhIB-EhlWTlBA-UEBRQFBA-U1FOUVJO-WVlZWQ=="

        Dim fromPdfPageNumber As Integer = Integer.Parse(textBoxFromPage.Text)
        Dim toPdfPageNumber As Integer = 0
        If textBoxToPage.Text.Length > 0 Then
            toPdfPageNumber = Integer.Parse(textBoxToPage.Text)
        End If

        ' search the text in PDF document
        Dim searchTextInstances As PdfTextSearchItem() = pdfTextExtractor.SearchText(pdfFile, textToSearch,
                    fromPdfPageNumber, toPdfPageNumber, checkBoxMatchCase.Checked, checkBoxMatchWholeWord.Checked)

        ' load the PDF file to search
        Dim pdfDocument As PdfDocument = pdfDocument.FromFile(pdfFile)

        ' set a demo serial number
        pdfDocument.SerialNumber = "YCgJMTAE-BiwJAhIB-EhlWTlBA-UEBRQFBA-U1FOUVJO-WVlZWQ=="

        ' highlight the searched text in PDF document
        For Each searchTextInstance As PdfTextSearchItem In searchTextInstances
            Dim pdfRectangle As PdfRectangle = New PdfRectangle(searchTextInstance.BoundingRectangle)

            ' set rectangle color and opacity
            pdfRectangle.BackColor = Color.Yellow
            pdfRectangle.Opacity = 30

            ' highlight the text
            pdfDocument.Pages(searchTextInstance.PdfPageNumber - 1).Layout(pdfRectangle)
        Next searchTextInstance

        ' write the modified PDF document
        Try
            ' write the PDF document to a memory buffer
            Dim pdfBuffer As Byte() = pdfDocument.WriteToMemory()

            ' inform the browser about the binary data format
            HttpContext.Current.Response.AddHeader("Content-Type", "application/pdf")

            ' let the browser know how to open the PDF document and the file name
            HttpContext.Current.Response.AddHeader("Content-Disposition", String.Format("attachment; filename=SearchText.pdf; size={0}",
                        pdfBuffer.Length.ToString()))

            ' write the PDF buffer to HTTP response
            HttpContext.Current.Response.BinaryWrite(pdfBuffer)

            ' call End() method of HTTP response to stop ASP.NET page processing
            HttpContext.Current.Response.End()
        Finally
            pdfDocument.Close()
        End Try
    End Sub

    Protected Sub Page_Load(ByVal sender As Object, ByVal e As System.EventArgs) Handles Me.Load
        If Not IsPostBack Then
            Dim pageUri As String = HttpContext.Current.Request.Url.AbsoluteUri
            hyperLinkOpenPdf.NavigateUrl = pageUri.Substring(0, pageUri.LastIndexOf("/"c)) + "/DemoFiles/Pdf/InputPdf.pdf"

            Master.SelectNode("searchText")
        End If
    End Sub
End Class
