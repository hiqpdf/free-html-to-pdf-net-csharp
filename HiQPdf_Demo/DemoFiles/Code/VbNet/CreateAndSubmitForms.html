Imports System.Configuration
Imports System.Collections
Imports System.Web.Security
Imports System.Web.UI
Imports System.Web.UI.WebControls
Imports System.Web.UI.WebControls.WebParts
Imports System.Web.UI.HtmlControls

Imports HiQPdf

Namespace HiQPdf_Demo
    Partial Public Class CreateAndSubmitForms
        Inherits System.Web.UI.Page
        Protected Sub buttonCreatePdf_Click(ByVal sender As Object, ByVal e As EventArgs) Handles buttonCreatePdf.Click
            ' create a PDF document
            Dim document As New PdfDocument()

            ' set a demo serial number
            document.SerialNumber = "YCgJMTAE-BiwJAhIB-EhlWTlBA-UEBRQFBA-U1FOUVJO-WVlZWQ=="

            ' add a page to document
            Dim page As PdfPage = document.AddPage()

            ' create true type fonts that can be used in document
            Dim ttfFont As New Font("Times New Roman", 10, GraphicsUnit.Point)
            Dim newTimesFont As PdfFont = document.CreateFont(ttfFont, False)

            ' create a standard font that can be used in document
            Dim helveticaStd As PdfFont = document.CreateStandardFont(PdfStandardFont.Helvetica)
            helveticaStd.Size = 10

            Dim crtXPos As Single = 10
            Dim crtYPos As Single = 10

            ' #Region "Add Check Box Field"

            If checkBoxAddCheckBox.Checked Then
                ' add a check box field to PDF form
                Dim checkBoxField As PdfFormCheckBox = document.Form.AddCheckBox(page, New RectangleF(crtXPos, crtYPos, 10, 10))

                checkBoxField.Checked = checkBoxCheckedState.Checked

                ' common field properties 
                checkBoxField.Name = "cb"
                checkBoxField.ToolTip = "Click to change the checked state"
                checkBoxField.Required = False
                checkBoxField.ReadOnly = False
                checkBoxField.Flatten = False

                ' advance the current drawing position in PDF page
                crtYPos = checkBoxField.BoundingRectangle.Bottom + 5
            End If

            ' #End Region

            ' #Region "Add Text Box Field"

            If checkBoxAddTextBox.Checked Then
                Dim initialText As String = textBoxInitialText.Text
                Dim textBoxField As PdfFormTextBox = document.Form.AddTextBox(page, New RectangleF(crtXPos, crtYPos, 300, 50), initialText, newTimesFont)

                textBoxField.IsMultiLine = checkBoxMultiline.Checked
                textBoxField.IsPassword = checkBoxIsPassword.Checked

                textBoxField.Style.ForeColor = Color.Navy
                textBoxField.Style.BackColor = Color.WhiteSmoke
                textBoxField.Style.BorderStyle = PdfBorderStyle.FixedSingle
                textBoxField.Style.BorderColor = Color.Green

                ' common field properties
                textBoxField.Name = "tb"
                textBoxField.ToolTip = "Please enter some text"
                textBoxField.Required = False
                textBoxField.ReadOnly = False
                textBoxField.DefaultValue = "Default text"
                textBoxField.Flatten = False

                ' advance the current drawing position in PDF page
                crtYPos = textBoxField.BoundingRectangle.Bottom + 5
            End If

            ' #End Region

            ' #Region "Add  List Box Field"

            If checkBoxAddListBox.Checked Then
                Dim listValues() As String = textBoxListBoxValues.Text.Split(New Char() {","c}, StringSplitOptions.RemoveEmptyEntries)

                Dim listBoxField As PdfFormListBox = document.Form.AddListBox(page, New RectangleF(crtXPos, crtYPos, 300, 50), listValues, helveticaStd)

                ' common field properties
                listBoxField.Name = "lb"
                listBoxField.ToolTip = "Select an element from the list"
                listBoxField.Required = False
                listBoxField.ReadOnly = False
                If (listValues.Length > 0) Then
                    listBoxField.DefaultValue = listValues(0)
                End If
                listBoxField.Flatten = False

                ' advance the current drawing position in PDF page
                crtYPos = listBoxField.BoundingRectangle.Bottom + 5
            End If

            ' #End Region

            ' #Region "Add Combo Box Field"

            If checkBoxAddComboBox.Checked Then
                Dim listValues() As String = textBoxComboBoxValues.Text.Split(New Char() {","c}, StringSplitOptions.RemoveEmptyEntries)

                Dim comboBoxField As PdfFormComboBox = document.Form.AddComboBox(page, New RectangleF(crtXPos, crtYPos, 300, 15), listValues, helveticaStd)

                comboBoxField.Editable = checkBoxEditableCombo.Checked

                ' common field properties
                comboBoxField.Name = "combo"
                comboBoxField.ToolTip = "Select an element from the combo drop down"
                comboBoxField.Required = False
                comboBoxField.ReadOnly = False
                If (listValues.Length > 0) Then
                    comboBoxField.DefaultValue = listValues(0)
                End If
                comboBoxField.Flatten = False

                ' advance the current drawing position in PDF page
                crtYPos = comboBoxField.BoundingRectangle.Bottom + 5
            End If

            ' #End Region

            ' #Region "Add Radio Buttons Group Field"

            If checkBoxAddRadioButtons.Checked Then
                Dim radioGroup As PdfFormRadioButtonsGroup = document.Form.AddRadioButtonsGroup(page)

                Dim rb1 As PdfFormRadioButton = radioGroup.AddRadioButton(New RectangleF(crtXPos, crtYPos, 10, 10), "rb1")
                Dim rb2 As PdfFormRadioButton = radioGroup.AddRadioButton(New RectangleF(crtXPos + 20, crtYPos, 10, 10), "rb2")

                radioGroup.SetCheckedRadioButton(rb2)

                radioGroup.Name = "rg"

                ' advance the current drawing position in PDF page
                crtYPos = rb1.BoundingRectangle.Bottom + 20
            End If

            ' #End Region

            ' #Region "Create the Submit Button"

            ' create the Submit button
            Dim submitButton As PdfFormButton = document.Form.AddButton(page, New RectangleF(crtXPos, crtYPos, 100, 20), "Submit Form", newTimesFont)
            submitButton.Name = "submitButton"

            ' create the submit action with the submit URL
            Dim submitAction As New PdfSubmitFormAction(textBoxSubmitUrl.Text)
            ' set the action flags such that the form values are submitted in HTML form format
            submitAction.Flags = submitAction.Flags Or PdfFormSubmitFlags.ExportFormat
            If radioButtonGet.Checked Then
                submitAction.Flags = submitAction.Flags Or PdfFormSubmitFlags.GetMethod
            End If

            ' set the submit button action
            submitButton.Action = submitAction

            ' #End Region

            ' #Region "Create the Reset Button"

            If checkBoxAddResetButton.Checked Then
                ' create the reset button
                Dim resetButton As PdfFormButton = document.Form.AddButton(page, New RectangleF(crtXPos + 120, crtYPos, 100, 20), "Reset Form", newTimesFont)
                resetButton.Name = "resetButton"

                ' create the reset action
                Dim resetAction As New PdfResetFormAction()

                ' set the reset button action
                resetButton.Action = resetAction
            End If

            ' #End Region

            Try
                ' write the PDF document to a memory buffer
                Dim pdfBuffer() As Byte = document.WriteToMemory()

                ' inform the browser about the binary data format
                HttpContext.Current.Response.AddHeader("Content-Type", "application/pdf")

                ' let the browser know how to open the PDF document and the file name
                HttpContext.Current.Response.AddHeader("Content-Disposition", String.Format("attachment; filename=CreateForms.pdf; size={0}", pdfBuffer.Length.ToString()))

                ' write the PDF buffer to HTTP response
                HttpContext.Current.Response.BinaryWrite(pdfBuffer)

                ' call End() method of HTTP response to stop ASP.NET page processing
                HttpContext.Current.Response.End()
            Finally
                document.Close()
            End Try
        End Sub

        Protected Sub Page_Load(ByVal sender As Object, ByVal e As EventArgs)
            If Not IsPostBack Then
                Master.SelectNode("createAndSubmitForms")
            End If
        End Sub
    End Class
End Namespace
