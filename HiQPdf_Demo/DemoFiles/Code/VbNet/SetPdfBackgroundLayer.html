Imports System.Configuration
Imports System.Collections
Imports System.Web.Security
Imports System.Web.UI
Imports System.Web.UI.WebControls
Imports System.Web.UI.WebControls.WebParts
Imports System.Web.UI.HtmlControls

Imports HiQPdf

Namespace HiQPdf_Demo
    Partial Public Class SetPdfBackgroundLayer
        Inherits System.Web.UI.Page
        Protected Sub buttonCreatePdf_Click(ByVal sender As Object, ByVal e As EventArgs)
            ' create the HTML to PDF converter
            Dim htmlToPdfConverter As New HtmlToPdf()

            ' set a demo serial number
            htmlToPdfConverter.SerialNumber = "YCgJMTAE-BiwJAhIB-EhlWTlBA-UEBRQFBA-U1FOUVJO-WVlZWQ=="

            ' attach to PageLayoutingEvent event raised right before layouting the HTML content in a PDF page
            AddHandler htmlToPdfConverter.PageLayoutingEvent, AddressOf htmlToPdfConverter_PageLayoutingEvent

            ' set PDF page margins
            htmlToPdfConverter.Document.Margins = New PdfMargins(Integer.Parse(textBoxLeftMargin.Text), Integer.Parse(textBoxRightMargin.Text), Integer.Parse(textBoxTopMargin.Text), Integer.Parse(textBoxBottomMargin.Text))

            Try
                Dim pdfBuffer() As Byte = htmlToPdfConverter.ConvertUrlToMemory(textBoxUrl.Text)

                ' inform the browser about the binary data format
                HttpContext.Current.Response.AddHeader("Content-Type", "application/pdf")

                ' let the browser know how to open the PDF document, attachment or inline, and the file name
                HttpContext.Current.Response.AddHeader("Content-Disposition", String.Format("attachment; filename=SetPdfBackground.pdf; size={0}", pdfBuffer.Length.ToString()))

                ' write the PDF buffer to HTTP response
                HttpContext.Current.Response.BinaryWrite(pdfBuffer)

                ' call End() method of HTTP response to stop ASP.NET page processing
                HttpContext.Current.Response.End()
            Finally
                ' dettach from PageLayoutingEvent event
                RemoveHandler htmlToPdfConverter.PageLayoutingEvent, AddressOf htmlToPdfConverter_PageLayoutingEvent
            End Try
        End Sub

        ''' <summary>
        ''' The PageLayoutingEvent event handler called before each PDF page is rendered by the converter
        ''' </summary>
        ''' <param name="eventParams">The event handler parameter giving information about the PDF page being rendered 
        ''' and the rectangle in page that will be rendered</param>
        Private Sub htmlToPdfConverter_PageLayoutingEvent(ByVal eventParams As PdfPageLayoutingParams)
            ' The PDF page being rendered
            Dim crtPage As PdfPage = eventParams.PdfPage

            ' draw a colored rectangle in the background of the PDF page
            Dim backColorRect As New PdfRectangle(0, 0, crtPage.DrawableRectangle.Width, crtPage.DrawableRectangle.Height)
            backColorRect.BackColor = Color.FromArgb(255, Integer.Parse(textBoxR.Text), Integer.Parse(textBoxG.Text), Integer.Parse(textBoxB.Text))
            crtPage.Layout(backColorRect)

            ' draw a 2 points orange line under the rendered content in page
            Dim leftBottom As New PointF(eventParams.LayoutingBounds.Left, eventParams.LayoutingBounds.Bottom + 1)
            Dim rightBottom As New PointF(eventParams.LayoutingBounds.Right, eventParams.LayoutingBounds.Bottom + 1)
            Dim bottomLine As New PdfLine(leftBottom, rightBottom)
            bottomLine.LineStyle.LineWidth = 2.0f
            bottomLine.ForeColor = Color.OrangeRed
            crtPage.Layout(bottomLine)
        End Sub

        Protected Sub Page_Load(ByVal sender As Object, ByVal e As EventArgs)
            If Not IsPostBack Then
                Master.SelectNode("setPdfBackgroundLayer")
            End If
        End Sub
    End Class
End Namespace
