Imports System.Web.UI
Imports System.Web.UI.WebControls

Imports HiQPdf

Namespace HiQPdf_Demo
    Partial Public Class PdfOutlinesAndLinks
        Inherits System.Web.UI.Page
        Protected Sub buttonCreatePdf_Click(ByVal sender As Object, ByVal e As EventArgs)
            ' create a PDF document
            Dim document As New PdfDocument()

            ' set a demo serial number
            document.SerialNumber = "YCgJMTAE-BiwJAhIB-EhlWTlBA-UEBRQFBA-U1FOUVJO-WVlZWQ=="

            ' display the outlines when the document is opened
            document.Viewer.PageMode = PdfPageMode.Outlines

            ' create the true type fonts that can be used in document
            Dim sysFont As New Font("Times New Roman", 10, GraphicsUnit.Point)
            Dim pdfFont As PdfFont = document.CreateFont(sysFont)
            Dim pdfFontEmbed As PdfFont = document.CreateFont(sysFont, True)

            Dim sysFontBold As New Font("Times New Roman", 12, FontStyle.Bold, GraphicsUnit.Point)
            Dim pdfFontBold As PdfFont = document.CreateFont(sysFontBold)
            Dim pdfFontBoldEmbed As PdfFont = document.CreateFont(sysFontBold, True)

            Dim sysFontBig As New Font("Times New Roman", 16, FontStyle.Bold, GraphicsUnit.Point)
            Dim pdfFontBig As PdfFont = document.CreateFont(sysFontBig)
            Dim pdfFontBigEmbed As PdfFont = document.CreateFont(sysFontBig, True)

            ' create a reference Graphics used for measuring
            Dim refBmp As New Bitmap(1, 1)
            Dim refGraphics As Graphics = Graphics.FromImage(refBmp)
            refGraphics.PageUnit = GraphicsUnit.Point

            ' create page 1
            Dim page1 As PdfPage = document.AddPage()

            Dim crtXPos As Single = 10
            Dim crtYPos As Single = 20

'			#Region "Create Table of contents"

            ' create table of contents title
            Dim tableOfContentsTitle As New PdfText(crtXPos, crtYPos, "Table of Contents", pdfFontBigEmbed)
            page1.Layout(tableOfContentsTitle)

            ' create a top outline for the table of contents
            Dim tableOfContentsDest As New PdfDestination(page1, New PointF(crtXPos, crtYPos))
            document.CreateTopOutline("Table of Contents", tableOfContentsDest)

            ' advance current Y position in page
            crtYPos += pdfFontBigEmbed.Size + 10

            ' create Chapter 1 in table of contents
            Dim chapter1TocTitle As New PdfText(crtXPos + 30, crtYPos, "Chapter 1", pdfFontBoldEmbed)
            ' layout the chapter 1 title in TOC
            page1.Layout(chapter1TocTitle)

            ' get the bounding rectangle of the chapter 1 title in TOC
            Dim textSize As SizeF = refGraphics.MeasureString("Chapter 1", sysFontBold)
            Dim chapter1TocTitleRectangle As New RectangleF(crtXPos + 30, crtYPos, textSize.Width, textSize.Height)

            ' advance current Y position in page
            crtYPos += pdfFontEmbed.Size + 10

            ' create Subchapter 1 in table of contents
            Dim subChapter11TocTitle As New PdfText(crtXPos + 60, crtYPos, "Subchapter 1", pdfFontEmbed)
            ' layout the text
            page1.Layout(subChapter11TocTitle)

            ' get the bounding rectangle of the subchapter 1 title in TOC
            textSize = refGraphics.MeasureString("Subchapter 1", sysFont)
            Dim subChapter11TocTitleRectangle As New RectangleF(crtXPos + 60, crtYPos, textSize.Width, textSize.Height)

            ' advance current Y position in page
            crtYPos += pdfFontEmbed.Size + 10

            ' create Subchapter 2 in table of contents
            Dim subChapter21TocTitle As New PdfText(crtXPos + 60, crtYPos, "Subchapter 2", pdfFontEmbed)
            ' layout the text
            page1.Layout(subChapter21TocTitle)

            ' get the bounding rectangle of the subchapter 2 title in TOC
            textSize = refGraphics.MeasureString("Subchapter 2", sysFont)
            Dim subChapter21TocTitleRectangle As New RectangleF(crtXPos + 60, crtYPos, textSize.Width, textSize.Height)

            ' advance current Y position in page
            crtYPos += pdfFontEmbed.Size + 10

            ' create Chapter 2 in table of contents
            Dim chapter2TocTitle As New PdfText(crtXPos + 30, crtYPos, "Chapter 2", pdfFontBoldEmbed)
            ' layout the text
            page1.Layout(chapter2TocTitle)

            ' get the bounding rectangle of the chapter 2 title in TOC
            textSize = refGraphics.MeasureString("Capter 2", sysFontBold)
            Dim chapter2TocTitleRectangle As New RectangleF(crtXPos + 30, crtYPos, textSize.Width, textSize.Height)

            ' advance current Y position in page
            crtYPos += pdfFontEmbed.Size + 10

            ' create Subchapter 1 in table of contents
            Dim subChapter12TocTitle As New PdfText(crtXPos + 60, crtYPos, "Subchapter 1", pdfFontEmbed)
            ' layout the text
            page1.Layout(subChapter12TocTitle)

            ' get the bounding rectangle of the subchapter 1 title in TOC
            textSize = refGraphics.MeasureString("Subchapter 1", sysFont)
            Dim subChapter12TocTitleRectangle As New RectangleF(crtXPos + 60, crtYPos, textSize.Width, textSize.Height)

            ' advance current Y position in page
            crtYPos += pdfFontEmbed.Size + 10

            ' create Subchapter 2 in table of contents
            Dim subChapter22TocTitle As New PdfText(crtXPos + 60, crtYPos, "Subchapter 2", pdfFontEmbed)
            ' layout the text
            page1.Layout(subChapter22TocTitle)

            ' get the bounding rectangle of the subchapter 2 title in TOC
            textSize = refGraphics.MeasureString("Subchapter 2", sysFont)
            Dim subChapter22TocTitleRectangle As New RectangleF(crtXPos + 60, crtYPos, textSize.Width, textSize.Height)

            ' advance current Y position in page
            crtYPos += pdfFontEmbed.Size + 10

            ' create the website link in the table of contents
            Dim visitWebSiteText As New PdfText(crtXPos + 30, crtYPos, "Visit HiQPdf Website", pdfFontEmbed)
            visitWebSiteText.ForeColor = Color.Navy
            ' layout the text
            page1.Layout(visitWebSiteText)

            ' get the bounding rectangle of the website link in TOC
            textSize = refGraphics.MeasureString("Visit HiQPdf Website", sysFont)
            Dim visitWebsiteRectangle As New RectangleF(crtXPos + 30, crtYPos, textSize.Width, textSize.Height)

            ' create the link to website in table of contents
            document.CreateUriLink(page1, visitWebsiteRectangle, "http://www.hiqpdf.com")

            ' advance current Y position in page
            crtYPos += pdfFontEmbed.Size + 10

            ' create a text note at the end of TOC
            Dim textNote As PdfTextNote = document.CreateTextNote(page1, New PointF(crtXPos + 10, crtYPos), "The table of contents contains internal links to chapters and subchapters")
            textNote.IconType = PdfTextNoteIconType.Note
            textNote.IsOpen = True

'			#End Region

'			#Region "Create Chapter 1 content and the link from TOC"

            ' create page 2
            Dim page2 As PdfPage = document.AddPage()

            ' create the Chapter 1 title
            Dim chapter1Title As New PdfText(crtXPos, 10, "Chapter 1", pdfFontBoldEmbed)
            ' layout the text
            page2.Layout(chapter1Title)

            ' create the Chapter 1 root outline
            Dim chapter1Destination As New PdfDestination(page2, New PointF(crtXPos, 10))
            Dim chapter1Outline As PdfOutline = document.CreateTopOutline("Chapter 1", chapter1Destination)
            chapter1Outline.TitleColor = Color.Navy

            ' create the PDF link from TOC to chapter 1
            document.CreatePdfLink(page1, chapter1TocTitleRectangle, chapter1Destination)

'			#End Region

'			#Region "Create Subchapter 1 content and the link from TOC"

            ' create the Subchapter 1 
            Dim subChapter11Title As New PdfText(crtXPos, 300, "Subchapter 1 of Chapter 1", pdfFontEmbed)
            ' layout the text
            page2.Layout(subChapter11Title)

            ' create subchapter 1 child outline
            Dim subChapter11Destination As New PdfDestination(page2, New PointF(crtXPos, 300))
            Dim subchapter11Outline As PdfOutline = document.CreateChildOutline("Subchapter 1", subChapter11Destination, chapter1Outline)

            ' create the PDF link from TOC to subchapter 1
            document.CreatePdfLink(page1, subChapter11TocTitleRectangle, subChapter11Destination)

'			#End Region

'			#Region "Create Subchapter 2 content and the link from TOC"

            ' create the Subchapter 2 
            Dim subChapter21Title As New PdfText(crtXPos, 600, "Subchapter 2 of Chapter 1", pdfFontEmbed)
            ' layout the text
            page2.Layout(subChapter21Title)

            ' create subchapter 2 child outline
            Dim subChapter21Destination As New PdfDestination(page2, New PointF(crtXPos, 600))
            Dim subchapter21Outline As PdfOutline = document.CreateChildOutline("Subchapter 2", subChapter21Destination, chapter1Outline)

            ' create the PDF link from TOC to subchapter 2
            document.CreatePdfLink(page1, subChapter21TocTitleRectangle, subChapter21Destination)

'			#End Region

'			#Region "Create Chapter 2 content and the link from TOC"

            ' create page 3
            Dim page3 As PdfPage = document.AddPage()

            ' create the Chapter 2 title
            Dim chapter2Title As New PdfText(crtXPos, 10, "Chapter 2", pdfFontBoldEmbed)
            ' layout the text
            page3.Layout(chapter2Title)

            ' create chapter 2 to outline
            Dim chapter2Destination As New PdfDestination(page3, New PointF(crtXPos, 10))
            Dim chapter2Outline As PdfOutline = document.CreateTopOutline("Chapter 2", chapter2Destination)
            chapter2Outline.TitleColor = Color.Green

            ' create the PDF link from TOC to chapter 2
            document.CreatePdfLink(page1, chapter2TocTitleRectangle, chapter2Destination)

'			#End Region

'			#Region "Create Subchapter 1 content and the link from TOC"

            ' create the Subchapter 1 
            Dim subChapter12Title As New PdfText(crtXPos, 300, "Subchapter 1 of Chapter 2", pdfFontEmbed)
            ' layout the text
            page3.Layout(subChapter12Title)

            ' create subchapter 1 child outline
            Dim subChapter12Destination As New PdfDestination(page3, New PointF(crtXPos, 300))
            Dim subchapter12Outline As PdfOutline = document.CreateChildOutline("Subchapter 1", subChapter12Destination, chapter2Outline)

            ' create the PDF link from TOC to subchapter 1
            document.CreatePdfLink(page1, subChapter12TocTitleRectangle, subChapter12Destination)

'			#End Region

'			#Region "Create Subchapter 2 content and the link from TOC"

            ' create the Subchapter 2 
            Dim subChapter22Title As New PdfText(crtXPos, 600, "Subchapter 2 of Chapter 2", pdfFontEmbed)
            ' layout the text
            page3.Layout(subChapter22Title)

            ' create subchapter 2 child outline
            Dim subChapter22Destination As New PdfDestination(page3, New PointF(crtXPos, 600))
            Dim subchapter22Outline As PdfOutline = document.CreateChildOutline("Subchapter 2", subChapter22Destination, chapter2Outline)

            ' create the PDF link from TOC to subchapter 2
            document.CreatePdfLink(page1, subChapter22TocTitleRectangle, subChapter22Destination)

'			#End Region

            ' dispose the graphics used for measuring
            refGraphics.Dispose()
            refBmp.Dispose()

            Try
                ' write the PDF document to a memory buffer
                Dim pdfBuffer() As Byte = document.WriteToMemory()

                ' inform the browser about the binary data format
                HttpContext.Current.Response.AddHeader("Content-Type", "application/pdf")

                ' let the browser know how to open the PDF document and the file name
                HttpContext.Current.Response.AddHeader("Content-Disposition", String.Format("attachment; filename=PdfOutlines.pdf; size={0}", pdfBuffer.Length.ToString()))

                ' write the PDF buffer to HTTP response
                HttpContext.Current.Response.BinaryWrite(pdfBuffer)

                ' call End() method of HTTP response to stop ASP.NET page processing
                HttpContext.Current.Response.End()
            Finally
                document.Close()
            End Try
        End Sub

        Protected Sub Page_Load(ByVal sender As Object, ByVal e As EventArgs)
            If Not IsPostBack Then
                Master.SelectNode("pdfOutlines")
            End If
        End Sub
    End Class
End Namespace
